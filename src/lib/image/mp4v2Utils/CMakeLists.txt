INCLUDE(cxx_defaults)

SET(_target
    "mp4v2Utils"
)

SET(_sources
    mp4v2Utils.cpp
)

ADD_LIBRARY(
  ${_target} STATIC
  ${_sources}
)
TARGET_INCLUDE_DIRECTORIES(
  ${_target}
  PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}
)

TARGET_LINK_LIBRARIES(
  ${_target}
  PRIVATE mp4v2
)

RV_STAGE(TYPE "LIBRARY" TARGET ${_target})
